#include <AccelStepper.h>
#include <MultiStepper.h>

int enablePin1 = 6;
int stepPin1 = 3;
int dirPin1 = 2;
int ms1Pin1 = 5;
int ms2Pin1 = 4;

int enablePin2 = 11;
int stepPin2 = 8;
int dirPin2 = 7;
int ms1Pin2 = 10;
int ms2Pin2 = 9;

AccelStepper motor1(1, stepPin1, dirPin1);
AccelStepper motor2(1, stepPin2, dirPin2);

void setup() {
  // put your setup code here, to run once:
 
  pinMode(ms1Pin1, OUTPUT);
  pinMode(ms2Pin1, OUTPUT);
  pinMode(stepPin1, OUTPUT);
  pinMode(dirPin1, OUTPUT);
  pinMode(enablePin1, OUTPUT);
  pinMode(ms1Pin2, OUTPUT);
  pinMode(ms2Pin2, OUTPUT);
  pinMode(stepPin2, OUTPUT);
  pinMode(dirPin2, OUTPUT);
  pinMode(enablePin2, OUTPUT);
 
 
  motor1.setMaxSpeed(10000);
  motor2.setMaxSpeed(10000);
}

void loop() {
  // put your main code here, to run repeatedly:
  digitalWrite(ms1Pin1, HIGH);
  digitalWrite(ms2Pin1, HIGH);
  digitalWrite(enablePin1, LOW);

  motor1.setSpeed(-10000);
  motor1.runSpeed();

  digitalWrite(ms1Pin2, HIGH);
  digitalWrite(ms2Pin2, HIGH);
  digitalWrite(enablePin2, LOW);

  motor2.setSpeed(-10000);
  motor2.runSpeed();
}
